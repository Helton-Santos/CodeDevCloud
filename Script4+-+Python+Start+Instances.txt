import boto3


def lambda_handler(event, context):

    # Listar as regioes / List regions
    ec2_client = boto3.client('ec2')
    regions = [region['RegionName']
               for region in ec2_client.describe_regions()['Regions']]

    # Buscar em todas as regioes / Searching in all regions
    for region in regions:
        ec2 = boto3.resource('ec2', region_name=region)

        print("Region:", region)
	# Observação importante relacionado a esse "For loop", se não quiser que faça a busca por todas as regiões, basta excluir essa parte do For loop até aqui e será feita apenas a busca na região onde está configurado o seu Lambda.

        # Filtrar somente as instancias Desligadas / Filter only disconnected instances
        instances = ec2.instances.filter(
            Filters=[{'Name': 'instance-state-name',
                      'Values': ['stopped']}])

        # Iniciar as Instancias / Starting the instances
        for instance in instances:
            instance.start()
            print('Started instance: ', instance.id)
